var playerElement = document.getElementById("player-wrapper"); var ytid = '4hME3n8Lkbs'; // with CC //var ytid = '9ZXTBHqG6CA'; // no CC var player = new Clappr.Player({ source: ytid, poster: 'https://i.ytimg.com/vi/'+ytid+'/hqdefault.jpg', height: 360, width: 640, plugins: [YoutubePlugin, YoutubePlugin], YoutubeVars : {"languageCode":"en"} }); player.attachTo(playerElement);
